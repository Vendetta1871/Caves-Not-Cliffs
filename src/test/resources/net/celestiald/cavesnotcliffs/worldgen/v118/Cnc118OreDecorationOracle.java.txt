public final class Cnc118OreDecorationOracle {
    private static final long[] SEEDS = {
        0L, 1L, -1L, 123456789L, Long.MIN_VALUE, Long.MAX_VALUE
    };
    private static final int[][] CHUNKS = {
        {0, 0}, {-1, 0}, {0, -1}, {-1, -1}, {1, -2}, {-33, 17}
    };
    private static final String[] BIOME_NAMES = {
        "plains", "dripstone_caves", "badlands", "meadow", "plains", "dripstone_caves"
    };
    private static List<FeatureSpec> features() {
        return Arrays.asList(

    public static void main(String[] args) throws Exception {
        SharedConstants.tryDetectVersion();
        Bootstrap.bootStrap();
        BuiltinRegistries.bootstrap();
        PrintWriter out = new PrintWriter(args[0], "UTF-8");
        out.println("# official Minecraft Java 1.18.2 scoped ordinary-ore decoration oracle");
        for (int caseIndex = 0; caseIndex < SEEDS.length; ++caseIndex) {
            long seed = SEEDS[caseIndex];
            int chunkX = CHUNKS[caseIndex][0];
            int chunkZ = CHUNKS[caseIndex][1];
            Holder<Biome> biome = BuiltinRegistries.BIOME.getHolderOrThrow(
                net.minecraft.resources.ResourceKey.create(Registry.BIOME_REGISTRY,
                    new net.minecraft.resources.ResourceLocation(BIOME_NAMES[caseIndex])));
            SparseWorld world = new SparseWorld(chunkX, chunkZ, biome);
            NoiseBasedChunkGenerator generator = new NoiseBasedChunkGenerator(
                BuiltinRegistries.STRUCTURE_SETS, BuiltinRegistries.NOISE,
                MultiNoiseBiomeSource.Preset.OVERWORLD.biomeSource(BuiltinRegistries.BIOME), seed,
                BuiltinRegistries.NOISE_GENERATOR_SETTINGS.getHolderOrThrow(
                    NoiseGeneratorSettings.OVERWORLD));
            WorldgenRandom random = new WorldgenRandom(new XoroshiroRandomSource(0L));
            long decorationSeed = random.setDecorationSeed(seed, chunkX << 4, chunkZ << 4);
            for (FeatureSpec feature : features()) {
                if (!feature.includes(BIOME_NAMES[caseIndex])) continue;
                random.setFeatureSeed(decorationSeed, feature.index, feature.step);
                PlacedFeature registered = feature.feature.value();
                List<net.minecraft.world.level.levelgen.placement.PlacementModifier> modifiers =
                    registered.placement();
                PlacedFeature withoutBiomeFilter = new PlacedFeature(feature.blockMatchFeature(),
                    modifiers.subList(0, modifiers.size() - 1));
                withoutBiomeFilter.place(world.level, generator, random,
                    new BlockPos(chunkX << 4, -64, chunkZ << 4));
            }
            Summary summary = world.summary();
            out.println("decorate\t" + seed + "\t" + chunkX + "\t" + chunkZ + "\t"
                + BIOME_NAMES[caseIndex] + "\t" + summary.total + "\t" + summary.counts
                + "\t" + summary.digest);
        }
        out.close();
    }

    private static FeatureSpec spec(Holder<PlacedFeature> feature, int step, int index) {
        return new FeatureSpec(feature, step, index);
    }

    private static final class FeatureSpec {
        final Holder<PlacedFeature> feature;
        final int step;
        final int index;

        FeatureSpec(Holder<PlacedFeature> feature, int step, int index) {
            this.feature = feature;
            this.step = step;
            this.index = index;
        }

        boolean includes(String biome) {
            if (step == 7 || index == 31) return biome.equals("meadow");
            if (index == 23) return biome.equals("dripstone_caves");
            if (index == 24) return !biome.equals("dripstone_caves");
            if (index == 27) return biome.equals("badlands");
            return true;
        }

        Holder<ConfiguredFeature<?, ?>> blockMatchFeature() {
            OreConfiguration original = (OreConfiguration) feature.value().feature().value().config();
            List<OreConfiguration.TargetBlockState> targets = new ArrayList<>();
            if (original.targetStates.size() == 1) {
                BlockState result = original.targetStates.get(0).state;
                targets.add(OreConfiguration.target(new BlockMatchTest(Blocks.STONE), result));
                targets.add(OreConfiguration.target(new BlockMatchTest(Blocks.DEEPSLATE), result));
                targets.add(OreConfiguration.target(new BlockMatchTest(Blocks.GRANITE), result));
                targets.add(OreConfiguration.target(new BlockMatchTest(Blocks.DIORITE), result));
                targets.add(OreConfiguration.target(new BlockMatchTest(Blocks.ANDESITE), result));
                targets.add(OreConfiguration.target(new BlockMatchTest(Blocks.TUFF), result));
            } else {
                BlockState shallow = original.targetStates.get(0).state;
                BlockState deep = original.targetStates.get(1).state;
                targets.add(OreConfiguration.target(new BlockMatchTest(Blocks.STONE), shallow));
                targets.add(OreConfiguration.target(new BlockMatchTest(Blocks.GRANITE), shallow));
                targets.add(OreConfiguration.target(new BlockMatchTest(Blocks.DIORITE), shallow));
                targets.add(OreConfiguration.target(new BlockMatchTest(Blocks.ANDESITE), shallow));
                targets.add(OreConfiguration.target(new BlockMatchTest(Blocks.DEEPSLATE), deep));
                targets.add(OreConfiguration.target(new BlockMatchTest(Blocks.TUFF), deep));
            }
            return Holder.direct(new ConfiguredFeature<>(Feature.ORE,
                new OreConfiguration(targets, original.size,
                    original.discardChanceOnAirExposure)));
        }
    }

    private static final class Summary {
        final int total;
        final String counts;
        final String digest;

        Summary(int total, String counts, String digest) {
            this.total = total;
            this.counts = counts;
            this.digest = digest;
        }
    }

    private static final class SparseWorld implements InvocationHandler {
        final Map<Long, ProtoChunk> chunks = new HashMap<>();
        final WorldGenLevel level;
        final Holder<Biome> biome;
        final int minChunkX;
        final int maxChunkX;
        final int minChunkZ;
        final int maxChunkZ;

        SparseWorld(int chunkX, int chunkZ, Holder<Biome> biome) {
            this.biome = biome;
            minChunkX = chunkX - 1;
            maxChunkX = chunkX + 1;
            minChunkZ = chunkZ - 1;
            maxChunkZ = chunkZ + 1;
            LevelHeightAccessor height = LevelHeightAccessor.create(-64, 384);
            for (int cx = minChunkX; cx <= maxChunkX; ++cx) {
                for (int cz = minChunkZ; cz <= maxChunkZ; ++cz) {
                    ProtoChunk chunk = new ProtoChunk(new ChunkPos(cx, cz), UpgradeData.EMPTY,
                        height, BuiltinRegistries.BIOME, null);
                    fill(chunk, cx, cz);
                    chunks.put(key(cx, cz), chunk);
                }
            }
            level = (WorldGenLevel) Proxy.newProxyInstance(getClass().getClassLoader(),
                new Class[] {WorldGenLevel.class}, this);
        }

        private static void fill(ProtoChunk chunk, int chunkX, int chunkZ) {
            for (int sectionIndex = 0; sectionIndex < 24; ++sectionIndex) {
                LevelChunkSection section = chunk.getSection(sectionIndex);
                int sectionY = sectionIndex - 4;
                for (int localX = 0; localX < 16; ++localX) {
                    int x = (chunkX << 4) + localX;
                    for (int localY = 0; localY < 16; ++localY) {
                        int y = (sectionY << 4) + localY;
                        BlockState base = y < 0 ? Blocks.DEEPSLATE.defaultBlockState()
                            : Blocks.STONE.defaultBlockState();
                        for (int localZ = 0; localZ < 16; ++localZ) {
                            int z = (chunkZ << 4) + localZ;
                            if (!cavity(x, y, z)) {
                                section.setBlockState(localX, localY, localZ, base, false);
                            }
                        }
                    }
                }
            }
        }

        @Override
        public Object invoke(Object proxy, Method method, Object[] args) {
            String name = method.getName();
            if (name.equals("getMinBuildHeight")) return -64;
            if (name.equals("getMaxBuildHeight")) return 320;
            if (name.equals("getHeight") && method.getParameterCount() == 0) return 384;
            if (name.equals("getHeight") && method.getParameterCount() == 3) return 319;
            if (name.equals("getSectionsCount")) return 24;
            if (name.equals("getSectionIndex")) return ((Integer) args[0] + 64) >> 4;
            if (name.equals("getMinSection")) return -4;
            if (name.equals("getMaxSection")) return 20;
            if (name.equals("getChunk")) {
                if (args[0] instanceof Integer) {
                    return chunks.get(key((Integer) args[0], (Integer) args[1]));
                }
                BlockPos pos = (BlockPos) args[0];
                return chunks.get(key(Math.floorDiv(pos.getX(), 16), Math.floorDiv(pos.getZ(), 16)));
            }
            if (name.equals("ensureCanWrite")) {
                BlockPos pos = (BlockPos) args[0];
                return chunks.containsKey(key(Math.floorDiv(pos.getX(), 16),
                    Math.floorDiv(pos.getZ(), 16)));
            }
            if (name.equals("isOutsideBuildHeight")) {
                int y = args[0] instanceof BlockPos ? ((BlockPos) args[0]).getY()
                    : (Integer) args[0];
                return y < -64 || y >= 320;
            }
            if (name.equals("getBlockState")) {
                BlockPos pos = (BlockPos) args[0];
                ProtoChunk chunk = chunks.get(key(Math.floorDiv(pos.getX(), 16),
                    Math.floorDiv(pos.getZ(), 16)));
                return chunk == null ? Blocks.AIR.defaultBlockState() : chunk.getBlockState(pos);
            }
            if (name.equals("getBiome")) return biome;
            if (name.equals("getSeed")) return 0L;
            if (method.getReturnType() == boolean.class) return false;
            if (method.getReturnType() == int.class) return 0;
            if (method.getReturnType() == long.class) return 0L;
            if (method.getReturnType() == float.class) return 0.0F;
            if (method.getReturnType() == double.class) return 0.0D;
            return null;
        }

        Summary summary() throws Exception {
            MessageDigest digest = MessageDigest.getInstance("SHA-256");
            TreeMap<String, Integer> counts = new TreeMap<>();
            int total = 0;
            for (int x = minChunkX << 4; x < (maxChunkX + 1) << 4; ++x) {
                for (int y = -64; y < 320; ++y) {
                    for (int z = minChunkZ << 4; z < (maxChunkZ + 1) << 4; ++z) {
                        BlockState state = chunks.get(key(Math.floorDiv(x, 16),
                            Math.floorDiv(z, 16))).getBlockState(new BlockPos(x, y, z));
                        BlockState base = cavity(x, y, z) ? Blocks.AIR.defaultBlockState()
                            : y < 0 ? Blocks.DEEPSLATE.defaultBlockState()
                            : Blocks.STONE.defaultBlockState();
                        if (state == base) continue;
                        String name = Registry.BLOCK.getKey(state.getBlock()).getPath();
                        counts.put(name, counts.getOrDefault(name, 0) + 1);
                        String row = x + "," + y + "," + z + "," + name + "\n";
                        digest.update(row.getBytes(StandardCharsets.UTF_8));
                        ++total;
                    }
                }
            }
            StringBuilder countString = new StringBuilder();
            for (Map.Entry<String, Integer> entry : counts.entrySet()) {
                if (countString.length() > 0) countString.append(';');
                countString.append(entry.getKey()).append('=').append(entry.getValue());
            }
            return new Summary(total, countString.toString(), hex(digest.digest()));
        }

        private static boolean cavity(int x, int y, int z) {
            long hash = (long) x * 73428767L ^ (long) y * 912931L
                ^ (long) z * 42317861L;
            return (hash & 127L) == 0L;
        }

        private static long key(int x, int z) {
            return (long) x << 32 ^ z & 0xFFFFFFFFL;
        }

        private static String hex(byte[] bytes) {
            StringBuilder result = new StringBuilder(bytes.length * 2);
            for (byte value : bytes) result.append(String.format("%02x", value & 0xFF));
            return result.toString();
        }
    }
