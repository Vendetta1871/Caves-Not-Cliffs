package net.minecraft.world.level.levelgen;

import com.mojang.serialization.Codec;
import java.io.File;
import java.io.PrintWriter;
import java.lang.reflect.Constructor;
import net.minecraft.SharedConstants;
import net.minecraft.core.Holder;
import net.minecraft.server.Bootstrap;
import net.minecraft.util.CubicSpline;
import net.minecraft.world.level.biome.TerrainShaper;
import net.minecraft.world.level.levelgen.synth.NormalNoise;

/** Emits raw-double density-function fixtures from Mojang's named 1.18.2 server classes. */
public final class Cnc118DensityFunctionsOracle {
    private static final long[] SEEDS = {
        0L, 1L, -1L, 123456789L, Long.MIN_VALUE, Long.MAX_VALUE
    };
    private static final int[][] POSITIONS = {
        {0, -64, 0}, {1, -63, -1}, {-17, -1, 31}, {32, 0, -33},
        {-4097, 127, 8193}, {33554433, 319, -33554434}
    };
    private static final double[] EDGE_VALUES = {
        Double.NEGATIVE_INFINITY, -2.0D, -1.0D, -0.75D, -0.5D,
        -Double.MIN_VALUE, -0.0D, 0.0D, Double.MIN_VALUE, 0.5D, 0.75D,
        1.0D, 2.0D, Double.POSITIVE_INFINITY, Double.NaN
    };

    private final PrintWriter out;
    private int records;

    private Cnc118DensityFunctionsOracle(PrintWriter out) {
        this.out = out;
    }

    public static void main(String[] args) throws Exception {
        SharedConstants.tryDetectVersion();
        Bootstrap.bootStrap();
        PrintWriter out = new PrintWriter(new File(args[0]), "UTF-8");
        out.println("# Generated by Cnc118DensityFunctionsOracle against official Minecraft Java 1.18.2 server classes");
        Cnc118DensityFunctionsOracle oracle = new Cnc118DensityFunctionsOracle(out);
        oracle.emitMapped();
        oracle.emitGradients();
        oracle.emitBinary();
        oracle.emitRanges();
        oracle.emitRarity();
        oracle.emitSlides();
        oracle.emitSpline();
        oracle.emitTerrainShaperSpline();
        oracle.emitNoiseFamily();
        out.println("count\t" + oracle.records);
        out.close();
    }

    private void emitMapped() {
        for (DensityFunctions.Mapped.Type type : DensityFunctions.Mapped.Type.values()) {
            for (double input : EDGE_VALUES) {
                DensityFunction function = DensityFunctions.map(
                    new Probe(input, -2.0D, 3.0D), type);
                emit("mapped\t" + type.name() + "\t" + bits(input)
                    + "\t" + bits(function.compute(context(0, 0, 0)))
                    + "\t" + bits(function.minValue()) + "\t" + bits(function.maxValue()));
            }
        }
    }

    private void emitGradients() {
        int[] yValues = {-65, -64, -63, -1, 0, 1, 318, 319, 320, 321};
        DensityFunction ascending = DensityFunctions.yClampedGradient(-64, 320, 1.5D, -1.5D);
        DensityFunction descending = DensityFunctions.yClampedGradient(-8, 8, -2.0D, 6.0D);
        for (int y : yValues) {
            emit("gradient\toverworld\t" + y + "\t" + bits(ascending.compute(context(0, y, 0)))
                + "\t" + bits(ascending.minValue()) + "\t" + bits(ascending.maxValue()));
            emit("gradient\treverse\t" + y + "\t" + bits(descending.compute(context(0, y, 0)))
                + "\t" + bits(descending.minValue()) + "\t" + bits(descending.maxValue()));
        }
    }

    private void emitBinary() {
        Object[][] cases = {
            {"ADD", 1.25D, -2.0D, 3.0D, -0.5D, -4.0D, 5.0D},
            {"MUL", 0.0D, -2.0D, 3.0D, Double.NaN, -4.0D, 5.0D},
            {"MUL", -0.0D, -2.0D, 3.0D, -7.0D, -4.0D, 5.0D},
            {"MUL", -2.0D, -3.0D, -1.0D, -4.0D, -5.0D, -2.0D},
            {"MUL", 2.0D, 1.0D, 3.0D, 4.0D, 2.0D, 5.0D},
            {"MIN", -10.0D, -10.0D, -10.0D, Double.NaN, -4.0D, 5.0D},
            {"MIN", 1.0D, -2.0D, 3.0D, -4.0D, -4.0D, 5.0D},
            {"MAX", 10.0D, 10.0D, 10.0D, Double.NaN, -4.0D, 5.0D},
            {"MAX", 1.0D, -2.0D, 3.0D, 4.0D, -4.0D, 5.0D}
        };
        for (Object[] entry : cases) {
            String type = (String) entry[0];
            Probe first = new Probe((Double) entry[1], (Double) entry[2], (Double) entry[3]);
            Probe second = new Probe((Double) entry[4], (Double) entry[5], (Double) entry[6]);
            DensityFunction function;
            if ("ADD".equals(type)) {
                function = DensityFunctions.add(first, second);
            } else if ("MUL".equals(type)) {
                function = DensityFunctions.mul(first, second);
            } else if ("MIN".equals(type)) {
                function = DensityFunctions.min(first, second);
            } else {
                function = DensityFunctions.max(first, second);
            }
            double result = function.compute(context(0, 0, 0));
            emit("binary\t" + type + "\t" + bits(first.value) + "\t" + bits(first.min)
                + "\t" + bits(first.max) + "\t" + bits(second.value) + "\t"
                + bits(second.min) + "\t" + bits(second.max) + "\t" + bits(result)
                + "\t" + bits(function.minValue()) + "\t" + bits(function.maxValue())
                + "\t" + second.computeCalls);
        }

        for (String type : new String[] {"ADD", "MUL"}) {
            DensityFunction input = new Probe(-2.5D, -4.0D, 3.0D);
            for (double constant : new double[] {-3.0D, -0.0D, 0.0D, 2.0D}) {
                DensityFunction function = "ADD".equals(type)
                    ? DensityFunctions.add(DensityFunctions.constant(constant), input)
                    : DensityFunctions.mul(DensityFunctions.constant(constant), input);
                emit("constant_binary\t" + type + "\t" + bits(constant) + "\t"
                    + bits(function.compute(context(0, 0, 0))) + "\t"
                    + bits(function.minValue()) + "\t" + bits(function.maxValue()));
            }
        }
    }

    private void emitRanges() {
        for (double input : EDGE_VALUES) {
            DensityFunction function = DensityFunctions.rangeChoice(
                new Probe(input, -2.0D, 2.0D), -0.5D, 0.5D,
                DensityFunctions.constant(11.0D), DensityFunctions.constant(-13.0D));
            emit("range\t" + bits(input) + "\t" + bits(function.compute(context(0, 0, 0)))
                + "\t" + bits(function.minValue()) + "\t" + bits(function.maxValue()));
        }
    }

    private void emitRarity() {
        double[] values = {
            Math.nextDown(-0.75D), -0.75D, Math.nextUp(-0.75D),
            Math.nextDown(-0.5D), -0.5D, Math.nextUp(-0.5D),
            Math.nextDown(0.0D), -0.0D, 0.0D, Math.nextUp(0.0D),
            Math.nextDown(0.5D), 0.5D, Math.nextUp(0.5D),
            Math.nextDown(0.75D), 0.75D, Math.nextUp(0.75D), Double.NaN
        };
        for (DensityFunctions.WeirdScaledSampler.RarityValueMapper mapper
                : DensityFunctions.WeirdScaledSampler.RarityValueMapper.values()) {
            for (double value : values) {
                emit("rarity\t" + mapper.name() + "\t" + bits(value)
                    + "\t" + bits(mapper.mapper.get(value)));
            }
        }
    }

    private void emitSlides() {
        for (boolean amplified : new boolean[] {false, true}) {
            NoiseSettings settings = NoiseSettings.overworldNoiseSettings(amplified);
            DensityFunction function = new DensityFunctions.Slide(settings,
                new Probe(0.625D, -2.0D, 3.0D));
            for (int y : new int[] {-72, -65, -64, -63, -56, -48, -40, 295, 303, 311, 319, 320}) {
                emit("slide\t" + amplified + "\t" + y + "\t"
                    + bits(function.compute(context(0, y, 0))) + "\t"
                    + bits(function.minValue()) + "\t" + bits(function.maxValue()));
            }
        }
    }

    private void emitSpline() {
        DensityFunction coordinateFunction = DensityFunctions.yClampedGradient(-64, 320,
            -1.25D, 1.75D);
        TerrainShaper.CoordinateCustom coordinate = new TerrainShaper.CoordinateCustom(
            new Holder.Direct<DensityFunction>(coordinateFunction));
        CubicSpline<TerrainShaper.PointCustom> spline = CubicSpline.builder(coordinate)
            .addPoint(-1.0F, -0.75F, 0.25F)
            .addPoint(-0.2F, 0.5F, -0.4F)
            .addPoint(0.6F, -0.1F, 0.8F)
            .addPoint(1.25F, 1.5F, -0.2F)
            .build();
        DensityFunction function = new DensityFunctions.Spline(spline, -0.5D, 1.0D);
        for (int y : new int[] {-64, -32, 0, 63, 127, 191, 255, 319, 320}) {
            emit("spline\t" + y + "\t" + bits(function.compute(context(0, y, 0)))
                + "\t" + bits(function.minValue()) + "\t" + bits(function.maxValue()));
        }
    }

    private void emitTerrainShaperSpline() {
        TerrainShaper shaper = TerrainShaper.overworld(false);
        for (DensityFunctions.TerrainShaperSpline.SplineType type
                : DensityFunctions.TerrainShaperSpline.SplineType.values()) {
            DensityFunction function = new DensityFunctions.TerrainShaperSpline(
                new Probe(-0.125D, -1.0D, 1.0D), new Probe(0.375D, -1.0D, 1.0D),
                new Probe(-0.69D, -1.0D, 1.0D), shaper, type, -0.81D, 8.0D);
            emit("terrain_spline\t" + type.name() + "\t"
                + bits(function.compute(context(0, 0, 0))) + "\t"
                + bits(function.minValue()) + "\t" + bits(function.maxValue()));
        }
    }

    private void emitNoiseFamily() throws Exception {
        for (long seed : SEEDS) {
            NormalNoise.NoiseParameters parameters = parameters();
            Holder<NormalNoise.NoiseParameters> holder =
                new Holder.Direct<NormalNoise.NoiseParameters>(parameters);
            for (int[] position : POSITIONS) {
                DensityFunction.FunctionContext context = context(position[0], position[1],
                    position[2]);
                emitNoise("noise", seed, position,
                    new DensityFunctions.Noise(holder, normal(seed), 0.25D, 0.5D), context);
                emitNoise("shift_a", seed, position,
                    new DensityFunctions.ShiftA(holder, normal(seed)), context);
                emitNoise("shift_b", seed, position,
                    new DensityFunctions.ShiftB(holder, normal(seed)), context);
                emitNoise("shift", seed, position,
                    makeShift(holder, normal(seed)), context);
                emitNoise("shifted", seed, position,
                    new DensityFunctions.ShiftedNoise(DensityFunctions.constant(0.25D),
                        DensityFunctions.constant(-0.5D), DensityFunctions.constant(0.75D),
                        0.25D, 0.5D, holder, normal(seed)), context);
                emitNoise("weird_type1", seed, position,
                    new DensityFunctions.WeirdScaledSampler(DensityFunctions.constant(-0.51D),
                        holder, normal(seed),
                        DensityFunctions.WeirdScaledSampler.RarityValueMapper.TYPE1), context);
                emitNoise("weird_type2", seed, position,
                    new DensityFunctions.WeirdScaledSampler(DensityFunctions.constant(0.75D),
                        holder, normal(seed),
                        DensityFunctions.WeirdScaledSampler.RarityValueMapper.TYPE2), context);
            }
        }
    }

    private void emitNoise(String kind, long seed, int[] position, DensityFunction function,
            DensityFunction.FunctionContext context) {
        emit("noise\t" + kind + "\t" + seed + "\t" + position[0] + "\t" + position[1]
            + "\t" + position[2] + "\t" + bits(function.compute(context)) + "\t"
            + bits(function.minValue()) + "\t" + bits(function.maxValue()));
    }

    private static NormalNoise.NoiseParameters parameters() {
        return new NormalNoise.NoiseParameters(-3, 1.0D, 0.5D, 0.0D, 1.0D);
    }

    private static NormalNoise normal(long seed) {
        return NormalNoise.create(new XoroshiroRandomSource(seed), parameters());
    }

    private static DensityFunction makeShift(Holder<NormalNoise.NoiseParameters> holder,
            NormalNoise noise) throws Exception {
        Class<?> type = Class.forName(
            "net.minecraft.world.level.levelgen.DensityFunctions$Shift");
        Constructor<?> constructor = type.getDeclaredConstructor(Holder.class, NormalNoise.class);
        constructor.setAccessible(true);
        return (DensityFunction) constructor.newInstance(holder, noise);
    }

    private static DensityFunction.FunctionContext context(int x, int y, int z) {
        return new DensityFunction.SinglePointContext(x, y, z);
    }

    private void emit(String line) {
        out.println(line);
        ++records;
    }

    private static String bits(double value) {
        return Long.toUnsignedString(Double.doubleToRawLongBits(value));
    }

    private static final class Probe implements DensityFunction.SimpleFunction {
        final double value;
        final double min;
        final double max;
        int computeCalls;

        Probe(double value, double min, double max) {
            this.value = value;
            this.min = min;
            this.max = max;
        }

        @Override
        public double compute(DensityFunction.FunctionContext context) {
            ++computeCalls;
            return value;
        }

        @Override
        public double minValue() {
            return min;
        }

        @Override
        public double maxValue() {
            return max;
        }

        @Override
        public Codec<? extends DensityFunction> codec() {
            return null;
        }
    }
}
