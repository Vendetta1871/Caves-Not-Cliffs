import java.io.File;
import java.io.PrintWriter;

/** Emits Overworld noise-setting fixtures from Mojang's obfuscated 1.18.2 server classes. */
public final class Cnc118NoiseSettingsOracle {
    private static final double[] VALUES = {-64.0D, -1.0D, -0.0D, 0.0D, 1.0D, 64.0D};
    private static final double[] Y_COORDINATES = {
        -72.5D, -64.0D, -63.9D, -1.1D, -0.9D, 0.0D,
        7.9D, 8.0D, 311.9D, 319.0D, 320.0D, 328.0D
    };

    public static void main(String[] args) throws Exception {
        ab.a();
        yv.a();
        PrintWriter out = new PrintWriter(new File(args[0]), "UTF-8");
        out.println("# Generated by Cnc118NoiseSettingsOracle against official Minecraft Java 1.18.2 server classes");
        emit(out, false, "normal");
        emit(out, true, "amplified");
        out.close();
    }

    private static void emit(PrintWriter out, boolean amplified, String name) {
        cug settings = cug.a(amplified);
        cuf sampling = settings.g();
        cuh top = settings.h();
        cuh bottom = settings.i();
        out.println("settings\t" + name
            + "\t" + settings.e() + "\t" + settings.f()
            + "\t" + bits(sampling.a()) + "\t" + bits(sampling.b())
            + "\t" + bits(sampling.c()) + "\t" + bits(sampling.d())
            + "\t" + bits(top.a()) + "\t" + top.b() + "\t" + top.c()
            + "\t" + bits(bottom.a()) + "\t" + bottom.b() + "\t" + bottom.c()
            + "\t" + settings.j() + "\t" + settings.k()
            + "\t" + settings.a() + "\t" + settings.b()
            + "\t" + settings.c() + "\t" + settings.d());
        for (double value : VALUES) {
            for (double y : Y_COORDINATES) {
                out.println("slide\t" + name + "\t" + value + "\t" + y
                    + "\t" + bits(cud.a(settings, value, y)));
            }
        }
    }

    private static String bits(double value) {
        return Long.toUnsignedString(Double.doubleToRawLongBits(value));
    }
}
