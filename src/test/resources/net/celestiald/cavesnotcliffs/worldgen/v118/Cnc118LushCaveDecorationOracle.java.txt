/** Emits final lush-cave feature digests from official mapped Java 1.18.2. */
public final class Cnc118LushCaveDecorationOracle {
    private static final long[] SEEDS = {
        0L, 1L, -1L, 123456789L, Long.MIN_VALUE, Long.MAX_VALUE
    };
    private static final int[][] CHUNKS = {
        {0, 0}, {-1, 0}, {0, -1}, {-1, -1}, {1, -2}, {-33, 17}
    };
    private static final String[] FEATURES = {
        "lush_caves_ceiling_vegetation", "cave_vines", "lush_caves_clay",
        "lush_caves_vegetation", "rooted_azalea_tree", "spore_blossom",
        "classic_vines_cave_feature"
    };
    private static final int[] INDICES = {22, 23, 24, 25, 26, 27, 28};

    public static void main(String[] args) throws Exception {
        SharedConstants.tryDetectVersion();
        Bootstrap.bootStrap();
        BuiltinRegistries.bootstrap();
        bindTags();
        PrintWriter out = new PrintWriter(args[0], "UTF-8");
        out.println("# official Minecraft Java 1.18.2 lush cave decoration digest");
        for (int index = 0; index < SEEDS.length; ++index) {
            long seed = SEEDS[index];
            int chunkX = CHUNKS[index][0];
            int chunkZ = CHUNKS[index][1];
            SparseWorld world = new SparseWorld(chunkX, chunkZ);
            NoiseBasedChunkGenerator generator = new NoiseBasedChunkGenerator(
                BuiltinRegistries.STRUCTURE_SETS, BuiltinRegistries.NOISE,
                MultiNoiseBiomeSource.Preset.OVERWORLD.biomeSource(BuiltinRegistries.BIOME),
                seed, BuiltinRegistries.NOISE_GENERATOR_SETTINGS.getHolderOrThrow(
                    NoiseGeneratorSettings.OVERWORLD));
            WorldgenRandom random = new WorldgenRandom(new XoroshiroRandomSource(0L));
            long decorationSeed = random.setDecorationSeed(seed,
                chunkX << 4, chunkZ << 4);
            StringBuilder flags = new StringBuilder();
            for (int feature = 0; feature < FEATURES.length; ++feature) {
                if (flags.length() > 0) flags.append(',');
                flags.append(place(FEATURES[feature], INDICES[feature], world,
                    generator, random, decorationSeed, chunkX, chunkZ));
            }
            Summary summary = world.summary();
            out.println("decorate\t" + seed + "\t" + chunkX + "\t" + chunkZ
                + "\t" + flags + "\t" + summary.total + "\t" + summary.counts
                + "\t" + summary.digest);
        }
        out.close();
    }

    private static boolean place(String id, int featureIndex, SparseWorld world,
            NoiseBasedChunkGenerator generator, WorldgenRandom random, long decorationSeed,
            int chunkX, int chunkZ) {
        PlacedFeature registered = BuiltinRegistries.PLACED_FEATURE.get(
            new net.minecraft.resources.ResourceLocation(id));
        List<PlacementModifier> modifiers = registered.placement();
        PlacedFeature withoutBiome = new PlacedFeature(registered.feature(),
            modifiers.subList(0, modifiers.size() - 1));
        random.setFeatureSeed(decorationSeed, featureIndex, 9);
        return withoutBiome.place(world.level, generator, random,
            new BlockPos(chunkX << 4, -64, chunkZ << 4));
    }

    private static void bindTags() {
        List<Block> baseStone = blocks(Blocks.STONE, Blocks.GRANITE, Blocks.DIORITE,
            Blocks.ANDESITE, Blocks.TUFF, Blocks.DEEPSLATE);
        List<Block> dirt = blocks(Blocks.DIRT, Blocks.GRASS_BLOCK, Blocks.PODZOL,
            Blocks.COARSE_DIRT, Blocks.MYCELIUM, Blocks.ROOTED_DIRT, Blocks.MOSS_BLOCK);
        List<Block> sand = blocks(Blocks.SAND, Blocks.RED_SAND);
        List<Block> terracotta = blocks(Blocks.TERRACOTTA, Blocks.WHITE_TERRACOTTA,
            Blocks.ORANGE_TERRACOTTA, Blocks.MAGENTA_TERRACOTTA,
            Blocks.LIGHT_BLUE_TERRACOTTA, Blocks.YELLOW_TERRACOTTA,
            Blocks.LIME_TERRACOTTA, Blocks.PINK_TERRACOTTA, Blocks.GRAY_TERRACOTTA,
            Blocks.LIGHT_GRAY_TERRACOTTA, Blocks.CYAN_TERRACOTTA,
            Blocks.PURPLE_TERRACOTTA, Blocks.BLUE_TERRACOTTA,
            Blocks.BROWN_TERRACOTTA, Blocks.GREEN_TERRACOTTA,
            Blocks.RED_TERRACOTTA, Blocks.BLACK_TERRACOTTA);
        List<Block> leaves = blocks(Blocks.JUNGLE_LEAVES, Blocks.OAK_LEAVES,
            Blocks.SPRUCE_LEAVES, Blocks.DARK_OAK_LEAVES, Blocks.ACACIA_LEAVES,
            Blocks.BIRCH_LEAVES, Blocks.AZALEA_LEAVES,
            Blocks.FLOWERING_AZALEA_LEAVES);
        List<Block> caveVines = blocks(Blocks.CAVE_VINES_PLANT, Blocks.CAVE_VINES);
        List<Block> moss = join(baseStone, caveVines, dirt);
        List<Block> lushGround = join(moss,
            blocks(Blocks.CLAY, Blocks.GRAVEL, Blocks.SAND));
        List<Block> rootReplaceable = join(baseStone, dirt, terracotta,
            blocks(Blocks.RED_SAND, Blocks.CLAY, Blocks.GRAVEL, Blocks.SAND,
                Blocks.SNOW_BLOCK, Blocks.POWDER_SNOW));
        List<Block> azaleaGrowsOn = join(dirt, sand, terracotta,
            blocks(Blocks.SNOW_BLOCK, Blocks.POWDER_SNOW));
        List<Block> replaceablePlants = blocks(Blocks.GRASS, Blocks.FERN,
            Blocks.DEAD_BUSH, Blocks.VINE, Blocks.GLOW_LICHEN, Blocks.SUNFLOWER,
            Blocks.LILAC, Blocks.ROSE_BUSH, Blocks.PEONY, Blocks.TALL_GRASS,
            Blocks.LARGE_FERN, Blocks.HANGING_ROOTS);
        List<Block> smallDripleaf = blocks(Blocks.CLAY, Blocks.MOSS_BLOCK);
        List<Block> bigDripleaf = join(smallDripleaf, dirt, blocks(Blocks.FARMLAND));

        Map<TagKey<Block>, List<Holder<Block>>> tags = new LinkedHashMap<>();
        put(tags, BlockTags.BASE_STONE_OVERWORLD, baseStone);
        put(tags, BlockTags.DIRT, dirt);
        put(tags, BlockTags.SAND, sand);
        put(tags, BlockTags.TERRACOTTA, terracotta);
        put(tags, BlockTags.LEAVES, leaves);
        put(tags, BlockTags.LOGS, blocks(Blocks.OAK_LOG));
        put(tags, BlockTags.CAVE_VINES, caveVines);
        put(tags, BlockTags.MOSS_REPLACEABLE, moss);
        put(tags, BlockTags.LUSH_GROUND_REPLACEABLE, lushGround);
        put(tags, BlockTags.AZALEA_ROOT_REPLACEABLE, rootReplaceable);
        put(tags, BlockTags.AZALEA_GROWS_ON, azaleaGrowsOn);
        put(tags, BlockTags.REPLACEABLE_PLANTS, replaceablePlants);
        put(tags, BlockTags.SMALL_DRIPLEAF_PLACEABLE, smallDripleaf);
        put(tags, BlockTags.BIG_DRIPLEAF_PLACEABLE, bigDripleaf);
        Registry.BLOCK.bindTags(tags);

        Map<TagKey<Fluid>, List<Holder<Fluid>>> fluids = new LinkedHashMap<>();
        fluids.put(FluidTags.WATER, fluids(Fluids.WATER, Fluids.FLOWING_WATER));
        fluids.put(FluidTags.LAVA, fluids(Fluids.LAVA, Fluids.FLOWING_LAVA));
        Registry.FLUID.bindTags(fluids);
    }

    private static List<Block> blocks(Block... values) {
        List<Block> result = new ArrayList<>();
        for (Block value : values) result.add(value);
        return result;
    }

    @SafeVarargs
    private static List<Block> join(List<Block>... lists) {
        List<Block> result = new ArrayList<>();
        for (List<Block> list : lists) result.addAll(list);
        return result;
    }

    private static void put(Map<TagKey<Block>, List<Holder<Block>>> tags,
            TagKey<Block> tag, List<Block> blocks) {
        List<Holder<Block>> holders = new ArrayList<>();
        for (Block block : blocks) holders.add(block.builtInRegistryHolder());
        tags.put(tag, holders);
    }

    private static List<Holder<Fluid>> fluids(Fluid... values) {
        List<Holder<Fluid>> result = new ArrayList<>();
        for (Fluid value : values) result.add(value.builtInRegistryHolder());
        return result;
    }

    private static final class Summary {
        final int total;
        final String counts;
        final String digest;

        Summary(int total, String counts, String digest) {
            this.total = total;
            this.counts = counts;
            this.digest = digest;
        }
    }

    private static final class SparseWorld implements InvocationHandler {
        final Map<Long, ProtoChunk> chunks = new HashMap<>();
        final WorldGenLevel level;
        final int minChunkX;
        final int maxChunkX;
        final int minChunkZ;
        final int maxChunkZ;
        final int centerChunkX;
        final int centerChunkZ;

        SparseWorld(int chunkX, int chunkZ) {
            centerChunkX = chunkX;
            centerChunkZ = chunkZ;
            minChunkX = chunkX - 2;
            maxChunkX = chunkX + 2;
            minChunkZ = chunkZ - 2;
            maxChunkZ = chunkZ + 2;
            LevelHeightAccessor height = LevelHeightAccessor.create(-64, 384);
            for (int cx = minChunkX; cx <= maxChunkX; ++cx) {
                for (int cz = minChunkZ; cz <= maxChunkZ; ++cz) {
                    ProtoChunk chunk = new ProtoChunk(new ChunkPos(cx, cz), UpgradeData.EMPTY,
                        height, BuiltinRegistries.BIOME, null);
                    fill(chunk, cx, cz);
                    chunks.put(key(cx, cz), chunk);
                }
            }
            level = (WorldGenLevel) Proxy.newProxyInstance(getClass().getClassLoader(),
                new Class[] {WorldGenLevel.class}, this);
        }

        private static void fill(ProtoChunk chunk, int chunkX, int chunkZ) {
            for (int sectionIndex = 0; sectionIndex < 24; ++sectionIndex) {
                LevelChunkSection section = chunk.getSection(sectionIndex);
                int sectionY = sectionIndex - 4;
                for (int localX = 0; localX < 16; ++localX) {
                    int x = (chunkX << 4) + localX;
                    for (int localY = 0; localY < 16; ++localY) {
                        int y = (sectionY << 4) + localY;
                        for (int localZ = 0; localZ < 16; ++localZ) {
                            int z = (chunkZ << 4) + localZ;
                            BlockState state = baseline(x, y, z);
                            if (!state.isAir()) {
                                section.setBlockState(localX, localY, localZ, state, false);
                            }
                        }
                    }
                }
            }
        }

        @Override
        @SuppressWarnings("unchecked")
        public Object invoke(Object proxy, Method method, Object[] args) {
            String name = method.getName();
            if (name.equals("getMinBuildHeight")) return -64;
            if (name.equals("getMaxBuildHeight")) return 320;
            if (name.equals("getHeight") && method.getParameterCount() == 0) return 384;
            if (name.equals("getHeight") && method.getParameterCount() == 3) return 320;
            if (name.equals("getSectionsCount")) return 24;
            if (name.equals("getSectionIndex")) return ((Integer) args[0] + 64) >> 4;
            if (name.equals("getMinSection")) return -4;
            if (name.equals("getMaxSection")) return 20;
            if (name.equals("getChunk")) {
                if (args[0] instanceof Integer) {
                    return chunks.get(key((Integer) args[0], (Integer) args[1]));
                }
                BlockPos pos = (BlockPos) args[0];
                return chunks.get(key(Math.floorDiv(pos.getX(), 16),
                    Math.floorDiv(pos.getZ(), 16)));
            }
            if (name.equals("ensureCanWrite")) return writable((BlockPos) args[0]);
            if (name.equals("isOutsideBuildHeight")) {
                int y = args[0] instanceof BlockPos ? ((BlockPos) args[0]).getY()
                    : (Integer) args[0];
                return y < -64 || y >= 320;
            }
            if (name.equals("getBlockState")) return state((BlockPos) args[0]);
            if (name.equals("getFluidState")) return state((BlockPos) args[0]).getFluidState();
            if (name.equals("isEmptyBlock")) return state((BlockPos) args[0]).isAir();
            if (name.equals("isStateAtPosition")) {
                return ((Predicate<BlockState>) args[1]).test(state((BlockPos) args[0]));
            }
            if (name.equals("isWaterAt")) {
                return state((BlockPos) args[0]).getFluidState().isSourceOfType(Fluids.WATER);
            }
            if (name.equals("setBlock")) {
                BlockPos pos = (BlockPos) args[0];
                ProtoChunk chunk = chunk(pos);
                if (!writable(pos) || chunk == null
                        || pos.getY() < -64 || pos.getY() >= 320) return false;
                chunk.setBlockState(pos, (BlockState) args[1], false);
                return true;
            }
            if (name.equals("getBiome")) {
                return BuiltinRegistries.BIOME.getHolderOrThrow(
                    net.minecraft.resources.ResourceKey.create(Registry.BIOME_REGISTRY,
                        new net.minecraft.resources.ResourceLocation("lush_caves")));
            }
            if (name.equals("getSeed")) return 0L;
            if (method.getReturnType() == boolean.class) return false;
            if (method.getReturnType() == int.class) return 0;
            if (method.getReturnType() == long.class) return 0L;
            if (method.getReturnType() == float.class) return 0.0F;
            if (method.getReturnType() == double.class) return 0.0D;
            return null;
        }

        private boolean contains(BlockPos pos) {
            return chunk(pos) != null && pos.getY() >= -64 && pos.getY() < 320;
        }

        private boolean writable(BlockPos pos) {
            int chunkX = Math.floorDiv(pos.getX(), 16);
            int chunkZ = Math.floorDiv(pos.getZ(), 16);
            return Math.abs(chunkX - centerChunkX) <= 1
                && Math.abs(chunkZ - centerChunkZ) <= 1;
        }

        private ProtoChunk chunk(BlockPos pos) {
            return chunks.get(key(Math.floorDiv(pos.getX(), 16),
                Math.floorDiv(pos.getZ(), 16)));
        }

        private BlockState state(BlockPos pos) {
            ProtoChunk chunk = chunk(pos);
            return chunk == null || pos.getY() < -64 || pos.getY() >= 320
                ? Blocks.AIR.defaultBlockState() : chunk.getBlockState(pos);
        }

        Summary summary() throws Exception {
            MessageDigest digest = MessageDigest.getInstance("SHA-256");
            TreeMap<String, Integer> counts = new TreeMap<>();
            int total = 0;
            for (int x = minChunkX << 4; x < (maxChunkX + 1) << 4; ++x) {
                for (int y = -64; y < 320; ++y) {
                    for (int z = minChunkZ << 4; z < (maxChunkZ + 1) << 4; ++z) {
                        BlockState state = state(new BlockPos(x, y, z));
                        if (state.equals(baseline(x, y, z))) continue;
                        String encoded = encode(state);
                        counts.put(encoded, counts.getOrDefault(encoded, 0) + 1);
                        digest.update((x + "," + y + "," + z + "," + encoded + "\n")
                            .getBytes(StandardCharsets.UTF_8));
                        ++total;
                    }
                }
            }
            StringBuilder countString = new StringBuilder();
            for (Map.Entry<String, Integer> entry : counts.entrySet()) {
                if (countString.length() > 0) countString.append(';');
                countString.append(entry.getKey()).append('=').append(entry.getValue());
            }
            return new Summary(total, countString.toString(), hex(digest.digest()));
        }

        private static String encode(BlockState state) {
            String name = Registry.BLOCK.getKey(state.getBlock()).getPath();
            if (state.is(Blocks.CAVE_VINES_PLANT)) {
                return name + "[berries=" + state.getValue(CaveVines.BERRIES) + "]";
            }
            if (state.is(Blocks.CAVE_VINES)) {
                return name + "[age=" + state.getValue(CaveVinesBlock.AGE)
                    + ",berries=" + state.getValue(CaveVines.BERRIES) + "]";
            }
            if (state.is(Blocks.SMALL_DRIPLEAF)) {
                return name + "[facing=" + state.getValue(BlockStateProperties.HORIZONTAL_FACING)
                    .getName() + ",half=" + state.getValue(BlockStateProperties.DOUBLE_BLOCK_HALF)
                    .getSerializedName() + ",waterlogged="
                    + state.getValue(BlockStateProperties.WATERLOGGED) + "]";
            }
            if (state.is(Blocks.BIG_DRIPLEAF) || state.is(Blocks.BIG_DRIPLEAF_STEM)) {
                return name + "[facing=" + state.getValue(BlockStateProperties.HORIZONTAL_FACING)
                    .getName() + ",waterlogged="
                    + state.getValue(BlockStateProperties.WATERLOGGED) + "]";
            }
            if (state.is(Blocks.TALL_GRASS)) {
                return name + "[half=" + state.getValue(BlockStateProperties.DOUBLE_BLOCK_HALF)
                    .getSerializedName() + "]";
            }
            if (state.is(Blocks.VINE)) {
                for (Direction direction : Direction.values()) {
                    if (direction != Direction.DOWN
                            && state.getValue(VineBlock.getPropertyForFace(direction))) {
                        return name + "[face=" + direction.getName() + "]";
                    }
                }
            }
            if (state.is(Blocks.HANGING_ROOTS)) {
                return name + "[waterlogged="
                    + state.getValue(BlockStateProperties.WATERLOGGED) + "]";
            }
            if (state.is(Blocks.AZALEA_LEAVES)
                    || state.is(Blocks.FLOWERING_AZALEA_LEAVES)) {
                return name + "[distance=" + state.getValue(BlockStateProperties.DISTANCE) + "]";
            }
            return name;
        }

        private static BlockState baseline(int x, int y, int z) {
            int phase = Math.floorMod(y + 64, 32);
            return phase >= 8 && phase <= 19
                ? Blocks.AIR.defaultBlockState() : Blocks.DIRT.defaultBlockState();
        }

        private static long key(int x, int z) {
            return (long) x << 32 ^ z & 0xFFFFFFFFL;
        }

        private static String hex(byte[] bytes) {
            StringBuilder result = new StringBuilder(bytes.length * 2);
            for (byte value : bytes) result.append(String.format("%02x", value & 0xFF));
            return result.toString();
        }
    }
