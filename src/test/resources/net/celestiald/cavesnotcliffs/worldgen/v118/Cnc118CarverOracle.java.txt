    private static void emitScaffold(PrintWriter out, long seed, int chunkX, int chunkZ)
            throws Exception {
        Environment environment = new Environment("normal", seed, chunkX, chunkZ, false);
        fillScaffold(environment.chunk, chunkX, chunkZ);
        FakeAquifer aquifer = new FakeAquifer();
        Run run = carve(environment, seed, aquifer);
        int changed = 0;
        int minChangedY = Integer.MAX_VALUE;
        int maxChangedY = Integer.MIN_VALUE;
        Map<String, Integer> changedCounts = new TreeMap<String, Integer>();
        int minX = chunkX << 4;
        int minZ = chunkZ << 4;
        for (int localX = 0; localX < 16; ++localX) {
            for (int y = -64; y < 320; ++y) {
                for (int localZ = 0; localZ < 16; ++localZ) {
                    int x = minX + localX;
                    int z = minZ + localZ;
                    String actual = name(environment.chunk.getBlockState(new BlockPos(x, y, z)));
                    String before = name(scaffold(x, y, z));
                    if (!actual.equals(before)) {
                        ++changed;
                        minChangedY = Math.min(minChangedY, y);
                        maxChangedY = Math.max(maxChangedY, y);
                        changedCounts.put(actual, changedCounts.getOrDefault(actual, 0) + 1);
                    }
                }
            }
        }
        out.println("scaffold\t" + seed + "\t" + chunkX + "\t" + chunkZ + "\t"
            + run.starts.size() + "\t" + run.startCounts() + "\t" + run.startHash()
            + "\t" + maskCardinality(run.mask) + "\t" + hashMask(run.mask) + "\t"
            + changed + "\t" + minChangedY + "\t" + maxChangedY + "\t"
            + encodeCounts(changedCounts) + "\t" + hashChunk(environment.chunk, chunkX, chunkZ)
            + "\t" + countPostprocessing(environment.chunk) + "\t"
            + hashPostprocessing(environment.chunk));
    }

    private static void emitTerrain(PrintWriter out, String profile, long seed, int chunkX,
            int chunkZ) throws Exception {
        Environment environment = new Environment(profile, seed, chunkX, chunkZ, true);
        Run run = carve(environment, seed, environment.noiseChunk.aquifer());
        out.println("terrain\t" + profile + "\t" + seed + "\t" + chunkX + "\t"
            + chunkZ + "\t" + run.starts.size() + "\t" + run.startCounts() + "\t"
            + run.startHash() + "\t" + maskCardinality(run.mask) + "\t"
            + hashMask(run.mask) + "\t" + hashChunk(environment.chunk, chunkX, chunkZ)
            + "\t" + encodeCounts(countChunk(environment.chunk, chunkX, chunkZ)));
    }

    private static Run carve(Environment environment, long seed, Aquifer aquifer) {
        WorldgenRandom random = new WorldgenRandom(new LegacyRandomSource(0L));
        CarvingMask mask = new CarvingMask(384, -64);
        CarvingContext context = new CarvingContext(environment.generator,
            RegistryAccess.BUILTIN.get(), environment.height, environment.noiseChunk);
        List<String> starts = new ArrayList<String>();
        ConfiguredWorldCarver<?>[] carvers = {
            Carvers.CAVE.value(), Carvers.CAVE_EXTRA_UNDERGROUND.value(), Carvers.CANYON.value()
        };
        for (int offsetX = -8; offsetX <= 8; ++offsetX) {
            for (int offsetZ = -8; offsetZ <= 8; ++offsetZ) {
                ChunkPos source = new ChunkPos(environment.chunkX + offsetX,
                    environment.chunkZ + offsetZ);
                for (int index = 0; index < carvers.length; ++index) {
                    random.setLargeFeatureSeed(seed + index, source.x, source.z);
                    if (carvers[index].isStartChunk(random)) {
                        starts.add(IDS[index] + ',' + index + ',' + source.x + ',' + source.z
                            + ',' + largeFeatureSeed(seed + index, source.x, source.z));
                        carvers[index].carve(context, environment.chunk,
                            environment.biomeManager::getBiome, random, aquifer, source, mask);
                    }
                }
            }
        }
        return new Run(starts, mask);
    }

    private static long largeFeatureSeed(long seed, int chunkX, int chunkZ) {
        Random random = new Random(seed);
        long xMultiplier = random.nextLong();
        long zMultiplier = random.nextLong();
        return (long) chunkX * xMultiplier ^ (long) chunkZ * zMultiplier ^ seed;
    }

    private static void fillScaffold(ProtoChunk chunk, int chunkX, int chunkZ) {
        int minX = chunkX << 4;
        int minZ = chunkZ << 4;
        for (int localX = 0; localX < 16; ++localX) {
            for (int y = -64; y < 320; ++y) {
                for (int localZ = 0; localZ < 16; ++localZ) {
                    int x = minX + localX;
                    int z = minZ + localZ;
                    chunk.setBlockState(new BlockPos(x, y, z), scaffold(x, y, z), false);
                }
            }
        }
    }

    private static BlockState scaffold(int x, int y, int z) {
        int selector = Math.floorMod((long) x * 73428767L ^ (long) y * 912931L
            ^ (long) z * 42317861L, 37);
        if (selector == 0) return Blocks.BEDROCK.defaultBlockState();
        if (selector == 1) return Blocks.WATER.defaultBlockState();
        if (selector == 2) return Blocks.TUFF.defaultBlockState();
        if (selector == 3) return Blocks.CALCITE.defaultBlockState();
        return y < 0 ? Blocks.DEEPSLATE.defaultBlockState() : Blocks.STONE.defaultBlockState();
    }

    private static String hashChunk(ProtoChunk chunk, int chunkX, int chunkZ) throws Exception {
        MessageDigest digest = MessageDigest.getInstance("SHA-256");
        int minX = chunkX << 4;
        int minZ = chunkZ << 4;
        for (int localX = 0; localX < 16; ++localX) {
            for (int y = -64; y < 320; ++y) {
                for (int localZ = 0; localZ < 16; ++localZ) {
                    String value = localX + "," + y + "," + localZ + ","
                        + name(chunk.getBlockState(new BlockPos(minX + localX, y,
                            minZ + localZ))) + "\n";
                    digest.update(value.getBytes(StandardCharsets.UTF_8));
                }
            }
        }
        return hex(digest.digest());
    }

    private static Map<String, Integer> countChunk(ProtoChunk chunk, int chunkX, int chunkZ) {
        Map<String, Integer> result = new TreeMap<String, Integer>();
        int minX = chunkX << 4;
        int minZ = chunkZ << 4;
        for (int localX = 0; localX < 16; ++localX) {
            for (int y = -64; y < 320; ++y) {
                for (int localZ = 0; localZ < 16; ++localZ) {
                    String material = name(chunk.getBlockState(new BlockPos(minX + localX, y,
                        minZ + localZ)));
                    result.put(material, result.getOrDefault(material, 0) + 1);
                }
            }
        }
        return result;
    }

    private static String hashMask(CarvingMask mask) throws Exception {
        MessageDigest digest = MessageDigest.getInstance("SHA-256");
        for (long value : mask.toArray()) updateLong(digest, value);
        return hex(digest.digest());
    }

    private static int maskCardinality(CarvingMask mask) {
        int result = 0;
        for (long value : mask.toArray()) result += Long.bitCount(value);
        return result;
    }

    private static int countPostprocessing(ProtoChunk chunk) {
        int count = 0;
        for (ShortList list : chunk.getPostProcessing()) if (list != null) count += list.size();
        return count;
    }

    private static String hashPostprocessing(ProtoChunk chunk) throws Exception {
        List<String> positions = new ArrayList<String>();
        ShortList[] sections = chunk.getPostProcessing();
        for (int section = 0; section < sections.length; ++section) {
            if (sections[section] == null) continue;
            int sectionY = chunk.getSectionYFromSectionIndex(section);
            for (short packed : sections[section]) {
                BlockPos position = ProtoChunk.unpackOffsetCoordinates(packed, sectionY,
                    chunk.getPos());
                positions.add(position.getX() + "," + position.getY() + "," + position.getZ());
            }
        }
        java.util.Collections.sort(positions);
        MessageDigest digest = MessageDigest.getInstance("SHA-256");
        for (String position : positions) {
            digest.update((position + "\n").getBytes(StandardCharsets.UTF_8));
        }
        return hex(digest.digest());
    }

    private static String encodeCounts(Map<String, Integer> counts) {
        StringBuilder result = new StringBuilder();
        for (Map.Entry<String, Integer> entry : counts.entrySet()) {
            if (result.length() > 0) result.append(',');
            result.append(entry.getKey()).append('=').append(entry.getValue());
        }
        return result.length() == 0 ? "-" : result.toString();
    }

    private static String name(BlockState state) {
        return Registry.BLOCK.getKey(state.getBlock()).getPath();
    }

    private static void updateLong(MessageDigest digest, long value) {
        for (int shift = 56; shift >= 0; shift -= 8) digest.update((byte) (value >>> shift));
    }

    private static String hex(byte[] bytes) {
        StringBuilder result = new StringBuilder();
        for (byte value : bytes) result.append(String.format("%02x", value & 255));
        return result.toString();
    }

    private static ResourceKey<NoiseGeneratorSettings> profileKey(String profile) {
        if ("large".equals(profile)) return NoiseGeneratorSettings.LARGE_BIOMES;
        if ("amplified".equals(profile)) return NoiseGeneratorSettings.AMPLIFIED;
        return NoiseGeneratorSettings.OVERWORLD;
    }

    private static final class Environment {
        final long seed;
        final int chunkX;
        final int chunkZ;
        final LevelHeightAccessor height = LevelHeightAccessor.create(-64, 384);
        final NoiseBasedChunkGenerator generator;
        final ProtoChunk chunk;
        final NoiseChunk noiseChunk;
        final BiomeManager biomeManager;

        Environment(String profile, long seed, int chunkX, int chunkZ, boolean terrain) {
            this.seed = seed;
            this.chunkX = chunkX;
            this.chunkZ = chunkZ;
            Holder<NoiseGeneratorSettings> holder = BuiltinRegistries.NOISE_GENERATOR_SETTINGS
                .getHolderOrThrow(profileKey(profile));
            NoiseGeneratorSettings settings = holder.value();
            MultiNoiseBiomeSource biomes = MultiNoiseBiomeSource.Preset.OVERWORLD
                .biomeSource(BuiltinRegistries.BIOME);
            generator = new NoiseBasedChunkGenerator(BuiltinRegistries.STRUCTURE_SETS,
                BuiltinRegistries.NOISE, biomes, seed, holder);
            chunk = new ProtoChunk(new ChunkPos(chunkX, chunkZ), UpgradeData.EMPTY, height,
                BuiltinRegistries.BIOME, null);
            if (terrain) {
                int minX = chunkX << 4;
                int minZ = chunkZ << 4;
                for (int localX = 0; localX < 16; ++localX) {
                    for (int localZ = 0; localZ < 16; ++localZ) {
                        int x = minX + localX;
                        int z = minZ + localZ;
                        NoiseColumn column = generator.getBaseColumn(x, z, height);
                        for (int y = -64; y < 320; ++y) {
                            BlockState state = column.getBlock(y);
                            if (!state.isAir()) {
                                chunk.setBlockState(new BlockPos(x, y, z), state, false);
                            }
                        }
                    }
                }
            }
            Aquifer.FluidStatus lava = new Aquifer.FluidStatus(-54,
                Blocks.LAVA.defaultBlockState());
            Aquifer.FluidStatus fluid = new Aquifer.FluidStatus(settings.seaLevel(),
                settings.defaultFluid());
            Aquifer.FluidPicker picker = (x, y, z) -> y < Math.min(-54, settings.seaLevel())
                ? lava : fluid;
            noiseChunk = NoiseChunk.forChunk(chunk, generator.router(),
                () -> DensityFunctions.BeardifierMarker.INSTANCE, settings, picker,
                Blender.empty());
            biomeManager = new BiomeManager(generator, BiomeManager.obfuscateSeed(seed));
            if (terrain) {
                SurfaceSystem surfaces = new SurfaceSystem(BuiltinRegistries.NOISE,
                    settings.defaultBlock(), settings.seaLevel(), seed,
                    settings.getRandomSource());
                surfaces.buildSurface(biomeManager, BuiltinRegistries.BIOME,
                    settings.useLegacyRandomSource(), new WorldGenerationContext(generator,
                        height), chunk, noiseChunk, settings.surfaceRule());
                for (ShortList list : chunk.getPostProcessing()) if (list != null) list.clear();
            }
        }
    }

    private static final class FakeAquifer implements Aquifer {
        private boolean schedule;

        @Override
        public BlockState computeSubstance(DensityFunction.FunctionContext context,
                double density) {
            int selector = Math.floorMod((long) context.blockX() * 912931L
                ^ (long) context.blockY() * 42317861L
                ^ (long) context.blockZ() * 73428767L, 17);
            schedule = selector == 1 || selector == 3 || selector == 5;
            if (selector == 0) return null;
            if (selector == 1 || selector == 2) return Blocks.WATER.defaultBlockState();
            if (selector == 3 || selector == 4) return Blocks.LAVA.defaultBlockState();
            return Blocks.AIR.defaultBlockState();
        }

        @Override
        public boolean shouldScheduleFluidUpdate() {
            return schedule;
        }
    }

    private static final class Run {
        final List<String> starts;
        final CarvingMask mask;

        Run(List<String> starts, CarvingMask mask) {
            this.starts = starts;
            this.mask = mask;
        }

        String startCounts() {
            Map<String, Integer> counts = new TreeMap<String, Integer>();
            for (String start : starts) {
                String id = start.substring(0, start.indexOf(','));
                counts.put(id, counts.getOrDefault(id, 0) + 1);
            }
            return encodeCounts(counts);
        }

        String startHash() throws Exception {
            MessageDigest digest = MessageDigest.getInstance("SHA-256");
            for (String start : starts) {
                digest.update((start + "\n").getBytes(StandardCharsets.UTF_8));
            }
            return hex(digest.digest());
        }
    }
