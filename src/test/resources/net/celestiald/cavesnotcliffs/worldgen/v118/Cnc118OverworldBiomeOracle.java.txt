import com.mojang.datafixers.util.Pair;
import java.io.File;
import java.io.PrintWriter;
import java.util.ArrayList;
import java.util.LinkedHashSet;
import java.util.List;
import java.util.Set;

/** Emits the complete ordered Overworld climate table from the obfuscated 1.18.2 server. */
public final class Cnc118OverworldBiomeOracle {
    private static final long[] BASE_A = {0L, 0L, 0L, 0L, 0L, 0L};
    private static final long[] BASE_B = {-4321L, 2789L, 3456L, -6789L, 5000L, 8765L};

    private final PrintWriter out;
    private final List<Pair> entries = new ArrayList<>();
    private cbz.c index;
    private int lookupIndex;

    private Cnc118OverworldBiomeOracle(PrintWriter out) {
        this.out = out;
    }

    public static void main(String[] args) throws Exception {
        ab.a();
        yv.a();
        PrintWriter out = new PrintWriter(new File(args[0]), "UTF-8");
        out.println("# Generated by Cnc118OverworldBiomeOracle against official Minecraft Java 1.18.2 server classes");
        Cnc118OverworldBiomeOracle oracle = new Cnc118OverworldBiomeOracle(out);
        oracle.buildTable();
        oracle.emitEntries();
        oracle.emitRepresentativeLookups();
        oracle.emitBoundaryLookups();
        oracle.emitCaveMatrix();
        oracle.emitScatter();
        out.println("lookup-count\t" + oracle.lookupIndex);
        out.close();
    }

    private void buildTable() {
        ccd builder = new ccd();
        builder.a(pair -> entries.add((Pair) pair));
        index = new cbz.c(entries);
    }

    private void emitEntries() {
        out.println("entry-count\t" + entries.size());
        for (int entryIndex = 0; entryIndex < entries.size(); ++entryIndex) {
            Pair pair = entries.get(entryIndex);
            cbz.d point = (cbz.d) pair.getFirst();
            out.println("entry\t" + entryIndex
                + parameter(point.b()) + parameter(point.c()) + parameter(point.d())
                + parameter(point.e()) + parameter(point.f()) + parameter(point.g())
                + "\t" + point.h() + "\t" + biome(pair.getSecond()));
        }
    }

    private void emitRepresentativeLookups() {
        for (Pair pair : entries) {
            cbz.d point = (cbz.d) pair.getFirst();
            emitLookup("entry-mid",
                midpoint(point.b()), midpoint(point.c()), midpoint(point.d()),
                midpoint(point.e()), midpoint(point.f()), midpoint(point.g()));
        }
    }

    private void emitBoundaryLookups() {
        for (int axis = 0; axis < 6; ++axis) {
            Set<Long> boundaries = new LinkedHashSet<>();
            for (Pair pair : entries) {
                cbz.b parameter = parameter((cbz.d) pair.getFirst(), axis);
                boundaries.add(parameter.a());
                boundaries.add(parameter.b());
            }
            for (long boundary : boundaries) {
                for (long adjacent : around(boundary)) {
                    long[] a = BASE_A.clone();
                    a[axis] = adjacent;
                    emitLookup("axis-" + axis, a);
                    long[] b = BASE_B.clone();
                    b[axis] = adjacent;
                    emitLookup("axis-" + axis, b);
                }
            }
        }
    }

    private void emitCaveMatrix() {
        long[] temperatures = {-10000L, -4500L, 0L, 5500L, 10000L};
        long[] humidities = {6999L, 7000L, 7001L, 10000L};
        long[] continentalnesses = {7999L, 8000L, 8001L, 10000L};
        long[] depths = {1999L, 2000L, 2001L, 8999L, 9000L, 9001L};
        long[] weirdnesses = {-10000L, -5666L, -500L, 0L, 500L, 5666L, 10000L};
        for (long temperature : temperatures) {
            for (long humidity : humidities) {
                for (long continentalness : continentalnesses) {
                    for (long depth : depths) {
                        for (long weirdness : weirdnesses) {
                            emitLookup("cave", temperature, humidity, continentalness,
                                -1000L, depth, weirdness);
                        }
                    }
                }
            }
        }
    }

    private void emitScatter() {
        long state = 0x6a09e667f3bcc909L;
        for (int sample = 0; sample < 2048; ++sample) {
            long[] target = new long[6];
            for (int axis = 0; axis < target.length; ++axis) {
                state = mix(state + 0x9e3779b97f4a7c15L);
                target[axis] = Math.floorMod(state, 30001L) - 15000L;
            }
            emitLookup("scatter", target);
        }
    }

    private void emitLookup(String kind, long... target) {
        cbz.h point = new cbz.h(target[0], target[1], target[2], target[3], target[4], target[5]);
        Object result = index.a(point);
        out.println("lookup\t" + lookupIndex++ + "\t" + kind
            + "\t" + target[0] + "\t" + target[1] + "\t" + target[2]
            + "\t" + target[3] + "\t" + target[4] + "\t" + target[5]
            + "\t" + biome(result));
    }

    private static cbz.b parameter(cbz.d point, int axis) {
        switch (axis) {
            case 0: return point.b();
            case 1: return point.c();
            case 2: return point.d();
            case 3: return point.e();
            case 4: return point.f();
            case 5: return point.g();
            default: throw new IllegalArgumentException("axis " + axis);
        }
    }

    private static String parameter(cbz.b parameter) {
        return "\t" + parameter.a() + "\t" + parameter.b();
    }

    private static long midpoint(cbz.b parameter) {
        return parameter.a() + (parameter.b() - parameter.a()) / 2L;
    }

    private static long[] around(long value) {
        if (value == Long.MIN_VALUE) {
            return new long[] {value, value + 1L};
        }
        if (value == Long.MAX_VALUE) {
            return new long[] {value - 1L, value};
        }
        return new long[] {value - 1L, value, value + 1L};
    }

    private static String biome(Object value) {
        return ((ys) value).a().toString();
    }

    private static long mix(long value) {
        value = (value ^ value >>> 30) * 0xbf58476d1ce4e5b9L;
        value = (value ^ value >>> 27) * 0x94d049bb133111ebL;
        return value ^ value >>> 31;
    }
}
